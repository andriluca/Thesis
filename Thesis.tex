%----------------------------------------------------------------------------

%(1) As presented in this template you can write:
%    \chapter{Title of the chapter}
%    *body of the chapter*

\addtocontents{toc}{\vspace{2em}} % Add a gap in the Contents, for aesthetics
\mainmatter % Begin numeric (1,2,3...) page numbering

% --------------------------------------------------------------------------

% --------------------------------------------------------------------------
